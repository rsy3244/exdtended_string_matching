\begin{frame}{文字クラス}
	\begin{block}{文字クラス}
		
		文字集合の部分集合を表す記号
		
		例 $[\text{a}\mathchar`-\text{d}] = \{\text{a},\text{b},\text{c},\text{d}\}$
	\end{block}
	\begin{block}{文字列照合問題への適用}
		\begin{itemize}
			\item パターンに文字クラスが含まれる
			\item テキストに文字クラスが含まれる
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}{パターンに文字クラスが含まれる場合}
	\begin{block}{基本アイデア}
			文字クラス中の文字に対応するビットマスクのビットを立てる
	\end{block}
	例 $P = [\text{C},\text{G}]\text{ATAT}$

	\begin{columns}
